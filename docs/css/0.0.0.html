1，padding && margin


@for $i from 1 through 6 {
  // padding
  .p-#{$i*5} {
    padding: #{$i * 5}px;
  }
  // padding-left
  .pl-#{$i*5} {
    padding-left: #{$i * 5}px;
  }
  // padding-right
  .pr-#{$i*5} {
    padding-right: #{$i * 5}px;
  }
  // padding-top
  .pt-#{$i*5} {
    padding-top: #{$i * 5}px;
  }
  // padding-bottom
  .pb-#{$i*5} {
    padding-bottom: #{$i * 5}px;
  }
  // 水平方向
  .px-#{$i*5} {
    padding-left: #{$i * 5}px;
    padding-right: #{$i * 5}px;
  }
  // 垂直方向
  .py-#{$i*5} {
    padding-top: #{$i * 5}px;
    padding-bottom: #{$i * 5}px;
  }

  // margin
  .m-#{$i*5} {
    margin: #{$i * 5}px;
  }
  // margin-left
  .ml-#{$i*5} {
    margin-left: #{$i * 5}px;
  }
  // margin-right
  .mr-#{$i*5} {
    margin-right: #{$i * 5}px;
  }
  // margin-top
  .mt-#{$i*5} {
    margin-top: #{$i * 5}px;
  }
  // margin-bottom
  .mb-#{$i*5} {
    margin-bottom: #{$i * 5}px;
  }

  // 水平方向
  .mx-#{$i*5} {
    margin-left: #{$i * 5}px;
    margin-right: #{$i * 5}px;
  }
  // 垂直方向
  .my-#{$i*5} {
    margin-top: #{$i * 5}px;
    margin-bottom: #{$i * 5}px;
  }
}
