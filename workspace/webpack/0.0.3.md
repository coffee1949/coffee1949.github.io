# webpack-dev-server

安装`webpack-dev-server`

```javascript

npm i -D webpack-dev-server
```

使用`webpack-dev-server`

```javascript

module.exports = {
  devServer: {
    // 主机地址
    host: "127.0.0.1",
    // 端口号
    port: 3100,
    open: true,
    hot: true,
    // 跨域设置代理
    proxy: {
      "/api": {
        target: "https://xxxx.com/api/",
        changeOrigin: true,
        pathRewrite: { "^/api": "/" },
      },
    },
    // 是否开启gzip压缩
    compress: true,
  },
};
```
